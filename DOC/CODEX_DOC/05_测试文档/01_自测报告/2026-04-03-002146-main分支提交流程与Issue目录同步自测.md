```python
import json, os, subprocess

env = os.environ.copy()
env['http_proxy'] = 'http://127.0.0.1:10809'
env['https_proxy'] = 'http://127.0.0.1:10809'
query = '''query { repository(owner:"wgwtest", name:"NovelStoryManager") { issues(first:30, orderBy:{field:CREATED_AT, direction:ASC}) { nodes { number body } } } }'''
raw = subprocess.check_output(['gh', 'api', 'graphql', '-f', f'query={query}'], env=env, text=True)
issues = json.loads(raw)['data']['repository']['issues']['nodes']
old_tokens = [
    'DOC/CODEX_DOC/设计稿/',
    'DOC/CODEX_DOC/开发计划/',
    'DOC/CODEX_DOC/验收清单/',
    'DOC/CODEX_DOC/自测报告/',
    'DOC/CODEX_DOC/会话交接/',
    '/home/wgw/CodexProject/NovelStoryManager/',
]
remaining = []
for issue in issues:
    hits = [token for token in old_tokens if token in issue['body']]
    if hits:
        remaining.append((issue['number'], hits))
print('remaining_count=', len(remaining))
for item in remaining:
    print(item)
PY`
